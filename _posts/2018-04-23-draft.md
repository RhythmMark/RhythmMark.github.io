from base64 import *

with open ('mix64.txt','rb') as f:
    rs = f.read()

def encode(str):
    if max(str) > 'a':
        return b64decode(str)
    if max(str) > 'G':
        return b32decode(str)
    else :
        return b16decode(str)

for i in range(10):
    rs = encode(rs)
